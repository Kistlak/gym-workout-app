One-time setup steps
1. supabase.com → sign up → New project (~2 min).
2. Project Settings → API Keys → copy the Project URL and the publishable key.
3. SQL Editor → paste the snippet below → Run.
4. Pick a Sync code and use the same one on every device.
5. Tap Connect & sync. 
create table if not exists ppl_kv (
  user_key text not null,
  k text not null,
  v jsonb,
  updated_at timestamptz default now(),
  primary key (user_key, k)
);
alter table ppl_kv enable row level security;
create policy "anon access" on ppl_kv
  for all to anon using (true) with check (true);
